async function generateStory(userText) { document.getElementById('status').innerText = "Demande en cours..."; const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${API_KEY}`; try { const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: "Raconte une histoire très courte pour enfant sur : " + userText }] }] }) }); const data = await response.json(); if (data.error) { // C'EST ICI QUE LE SECRET SERA RÉVÉLÉ console.error(data.error); alert("MESSAGE DE GOOGLE : " + data.error.message); document.getElementById('status').innerText = "Erreur : " + data.error.status; return; } if (data.candidates && data.candidates[0].content) { currentStory = data.candidates[0].content.parts[0].text; displayResult(currentStory, userText); } } catch (e) { alert("Erreur réseau : " + e.message); } }
